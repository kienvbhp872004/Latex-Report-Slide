\section{Giới thiệu tổng quan}
% ============================================================

\begin{frame}{Image Fusion và Image Decomposition}
    \textbf{Image Fusion} là bài toán kết hợp thông tin từ nhiều ảnh nguồn khác modality (IR + VIS, MRI + CT, multifocus...) nhằm tạo ra \textbf{một ảnh duy nhất} chứa đầy đủ thông tin bổ sung.

    \vspace{0.2cm}
    \textbf{Ứng dụng tiêu biểu:} Giám sát an ninh, chẩn đoán y tế, xe tự lái, viễn thám, robotics.

    \vspace{0.3cm}
    \textbf{Image Decomposition} là thành phần \textbf{then chốt} trong hầu hết pipeline fusion SOTA.

    \vspace{0.2cm}
    \begin{block}{Ý tưởng cốt lõi}
        Phân tách ảnh nguồn thành các thành phần độc lập $\to$ áp dụng chiến lược fusion tối ưu cho từng thành phần $\to$ tổng hợp lại ảnh kết quả.
    \end{block}
\end{frame}

\begin{frame}{Tại sao cần Decomposition?}
    Trong fusion, hai loại thông tin cần xử lý khác nhau:

    \vspace{0.2cm}
    \begin{itemize}
        \item \textbf{Base / Low-frequency:} Thông tin cường độ nền, cấu trúc toàn cục (illumination, large-scale structure). \\
        $\Rightarrow$ Yêu cầu tích hợp nhất quán, tránh artifact.
        \item \textbf{Detail / High-frequency:} Cạnh, kết cấu, gradient cục bộ, nhiễu. \\
        $\Rightarrow$ Yêu cầu bảo toàn và làm sắc nét thông tin chi tiết từng modality.
    \end{itemize}

    \vspace{0.3cm}
    \begin{alertblock}{Vấn đề nếu không decomposition}
        Nếu áp dụng fusion rule đồng nhất cho toàn bộ ảnh $\Rightarrow$ mờ cạnh, mất chi tiết texture, hoặc sai lệch màu sắc.
    \end{alertblock}

    \vspace{0.2cm}
    \textbf{Decomposition} cho phép thiết kế \textbf{fusion rule riêng biệt} và phù hợp cho từng thành phần.
\end{frame}

    \end{block}
\end{frame}

\section{Nhóm 1: Multi-scale Transform Decomposition}
% ============================================================

\begin{frame}{Nhóm 1: Multi-scale Transform -- Tổng quan}
    \textbf{Đặc điểm:} Nhóm phương pháp truyền thống với nền tảng toán học chặt chẽ.

    \vspace{0.2cm}
    Sử dụng các biến đổi toán học để phân tách ảnh thành các \textbf{dải tần số} hoặc \textbf{tỉ lệ (scale)} khác nhau.

    \vspace{0.3cm}
    \textbf{Các phương pháp chính:}
    \begin{enumerate}
        \item Discrete Wavelet Transform (DWT)
        \item Non-subsampled Shearlet Transform (NSST)
        \item Gaussian / Laplacian Pyramid
        \item Guided Filter \& Edge-Preserving Decomposition
        \item Wavelet + Mamba: WMambaFuse (2025)
    \end{enumerate}
\end{frame}

\subsection{Discrete Wavelet Transform (DWT)}

\begin{frame}{DWT -- Nền tảng lý thuyết}
    \small
    DWT phân tách tín hiệu 2D bằng cách áp dụng bộ lọc \textbf{low-pass} $h[n]$ và \textbf{high-pass} $g[n]$ theo cả hai chiều (hàng và cột), sau đó \textbf{downsample 2x}.

    \vspace{0.15cm}
    Cho ảnh $I(x,y)$, một cấp DWT tạo ra \textbf{4 sub-band}:
    \begin{itemize}\setlength{\itemsep}{1pt}
        \item \textbf{LL:} Xấp xỉ (approximation) -- low-frequency, global structure
        \item \textbf{LH:} Chi tiết nằm ngang \quad \textbf{HL:} Chi tiết dọc
        \item \textbf{HH:} Chi tiết chéo -- góc, kết cấu mịn
    \end{itemize}

    \vspace{0.1cm}
    \begin{block}{Forward DWT cấp 1}
        \begin{align}
            c_{LL}(m,n) &= \sum_k \sum_l h[k]\, h[l] \cdot I(2m-k,\, 2n-l) \nonumber \\
            c_{LH}(m,n) &= \sum_k \sum_l g[k]\, h[l] \cdot I(2m-k,\, 2n-l)
        \end{align}
    \end{block}
\end{frame}

\begin{frame}{DWT -- Quy trình Fusion}
    \small
    \textbf{IDWT:} Tái tạo ảnh \textbf{không mất thông tin} (perfect reconstruction) khi bộ lọc thỏa điều kiện biorthogonal.

    \vspace{0.15cm}
    \textbf{Quy trình Fusion với DWT:}
    \begin{enumerate}\setlength{\itemsep}{2pt}
        \item DWT cho từng ảnh nguồn $I_1, I_2$ $\to$ $\{LL, LH, HL, HH\}$
        \item Fusion sub-band LL (low-freq): \textbf{weighted average}
        \item Fusion sub-band LH/HL/HH (high-freq): \textbf{max-absolute rule}:
        \vspace{-0.1cm}
        \begin{equation}
            F_{HH}(m,n) = \arg\max_{k \in \{1,2\}} |c_{HH}^k(m,n)|
        \end{equation}
        \vspace{-0.2cm}
        \item IDWT trên các sub-band đã fusion $\to$ ảnh kết quả
    \end{enumerate}

    \vspace{0.1cm}
    \begin{exampleblock}{\footnotesize Biến thể SOTA: WaveFusionNet (2024)}
        \footnotesize Tích hợp DWT vào encoder-decoder: DWT thay max-pooling. Feature wavelet feed vào attention module để tự động học fusion weight.
    \end{exampleblock}
\end{frame}

\begin{frame}{DWT -- Ưu nhược điểm}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Ưu điểm:}
            \begin{itemize}
                \item Perfect reconstruction
                \item Nền tảng lý thuyết chặt chẽ
                \item Tính toán nhanh
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{Nhược điểm:}
            \begin{itemize}
                \item \textbf{Shift-variant} (không bất biến dịch chuyển)
                \item Ít hướng: chỉ 3 hướng (H, V, D)
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\subsection{Non-subsampled Shearlet Transform (NSST)}

\begin{frame}{NSST -- Lý thuyết}
    \small
    \textbf{Động lực:} DWT chỉ có 3 hướng. NSST kết hợp NSCT và Shearlet để đạt phân giải \textbf{đa hướng, đa tỉ lệ} mà \textbf{không aliasing} (nhờ không downsample).

    \vspace{0.15cm}
    \begin{block}{Định nghĩa Shearlet}
        \vspace{-0.2cm}
        \begin{equation}
            \psi_{j,l,k}(x) = a^{3j/4}\, \psi\!\left(S_l\, A_a^j\, x - k\right)
        \end{equation}
        \vspace{-0.2cm}
        \footnotesize $A_a$: ma trận scale anisotropic \quad $S_l$: ma trận shear \quad $k$: vector dịch chuyển
    \end{block}

    \vspace{0.1cm}
    NSST \textbf{không downsample} $\Rightarrow$ sub-band cùng kích thước ảnh gốc $\Rightarrow$ dễ thiết kế fusion rule tại từng pixel.

    \vspace{0.1cm}
    \begin{exampleblock}{Ứng dụng}
        \footnotesize NSST đặc biệt hiệu quả trong \textbf{Medical Image Fusion} (MRI--CT, PET--MRI) nhờ khả năng bắt cạnh đa hướng.
    \end{exampleblock}
\end{frame}

\subsection{Gaussian / Laplacian Pyramid}

\begin{frame}{Gaussian / Laplacian Pyramid}
    \small
    \textbf{Gaussian Pyramid} tạo chuỗi ảnh mờ dần: $G_0 \supset G_1 \supset \ldots \supset G_N$. Mỗi cấp thu nhỏ bằng Gaussian kernel rồi downsample 2x:
    \vspace{-0.1cm}
    \begin{equation}
        G_{l+1}(i,j) = \sum_{m,n} w(m,n) \cdot G_l(2i+m,\, 2j+n)
    \end{equation}

    \vspace{-0.1cm}
    \textbf{Laplacian Pyramid} là sự khác biệt giữa các cấp Gaussian:
    \vspace{-0.1cm}
    \begin{equation}
        L_l = G_l - \text{EXPAND}(G_{l+1})
    \end{equation}
    \vspace{-0.2cm}
    \begin{itemize}\setlength{\itemsep}{1pt}
        \item EXPAND: upsample + blur. Mỗi $L_l$ chứa detail tại tỉ lệ tương ứng.
        \item $G_N$ (cấp thấp nhất) là residual base image.
    \end{itemize}

    \vspace{0.1cm}
    \begin{exampleblock}{\footnotesize Octave Gaussian Pyramid (2024)}
        \footnotesize Chia pyramid thành octave và interval. Dùng \textbf{guided filter} để tăng sharpness, giảm halo artifact.
    \end{exampleblock}
\end{frame}

\subsection{Guided Filter \& Edge-Preserving}

\begin{frame}{Guided Filter}
    \small
    \textbf{Guided Filter} (He et al., 2013) -- mô hình \textbf{tuyến tính cục bộ}:
    \vspace{-0.1cm}
    \begin{equation}
        q_i = a_k \cdot p_i + b_k, \quad \forall\, i \in \omega_k
    \end{equation}
    \vspace{-0.15cm}
    ($p_i$: input, $q_i$: output, $\omega_k$: cửa sổ cục bộ tại $k$)

    \vspace{0.1cm}
    Hệ số $(a_k, b_k)$ tối thiểu hóa:
    \vspace{-0.1cm}
    \begin{equation}
        E(a_k, b_k) = \sum_{i \in \omega_k} \left[(a_k G_i + b_k - p_i)^2 + \epsilon \cdot a_k^2\right]
    \end{equation}
    \vspace{-0.15cm}
    Guided image $G$ giúp bộ lọc \textbf{bảo toàn cạnh mạnh}.

    \vspace{0.1cm}
    \begin{exampleblock}{\footnotesize MLGEPF (2024)}
        \footnotesize Multilevel Guided Edge-Preserving Filter: multi-level decomposition nhanh $O(n)$. Phù hợp ảnh y tế (PET--MRI, CT--MRI).
    \end{exampleblock}
\end{frame}

\subsection{WMambaFuse (2025)}

\begin{frame}{Wavelet + Mamba: WMambaFuse (2025)}
    WMambaFuse kết hợp \textbf{hai module song song}:
    \begin{itemize}
        \item \textbf{WMFM} (Wavelet-domain Mamba Fusion Module): xử lý thông tin frequency
        \item \textbf{SAFM} (Spatial Attention Fusion Module): xử lý không gian
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Kiến trúc:}
    \begin{itemize}
        \item Swin Transformer encoder trích xuất feature đa tỉ lệ
        \item \textbf{Mamba (SSM)} được dùng thay vì self-attention truyền thống $O(n^2)$ để xử lý chuẩn token dài hiệu quả hơn ($O(n)$)
    \end{itemize}

    \vspace{0.3cm}
    \begin{block}{Xu hướng 2024--2025}
        Thay Transformer bằng Mamba/SSM trong frequency domain để giảm chi phí tính toán trên ảnh high-resolution.
    \end{block}
\end{frame}

\section{Nhóm 2: Feature Decomposition (Deep Learning)}
% ============================================================

\begin{frame}{Nhóm 2: Feature Decomposition -- Tổng quan}
    \textbf{Đặc điểm:} Học cách phân tách \textbf{trực tiếp từ dữ liệu} thay vì dùng transform cố định.

    \vspace{0.2cm}
    Mạng neural tự tìm ra biểu diễn tối ưu cho fusion task.

    \vspace{0.3cm}
    \textbf{Các phương pháp chính:}
    \begin{enumerate}
        \item CDDFuse -- Base-Detail Decomposition (CVPR 2023)
        \item Modality-Specific / Shared Decomposition (DAF-Net, 2024)
        \item Semantic Layer Decomposition (PSFusion, 2023)
        \item Transformer-CNN Hybrid (DAF-Net, 2024)
    \end{enumerate}
\end{frame}

\subsection{CDDFuse (CVPR 2023)}

\begin{frame}{CDDFuse -- Kiến trúc}
    \textbf{CDDFuse} (Correlation-Driven Dual-Branch Fusion) -- một trong những mô hình \textbf{SOTA quan trọng nhất 2023}.

    \vspace{0.2cm}
    \textbf{Kiến trúc gồm:}
    \begin{itemize}
        \item \textbf{Lite Transformer (LT) Block:} Xử lý low-frequency global feature (base). \\
        Transformer học mối quan hệ tầm xa, phù hợp để nắm bắt cấu trúc toàn cục.
        \item \textbf{Invertible Neural Network (INN) Block:} Xử lý high-frequency local detail. \\
        INN đảm bảo thông tin detail không bị mất nhờ tính khả nghịch.
        \item \textbf{Correlation-Driven Loss:} Điều khiển quá trình học decomposition.
    \end{itemize}
\end{frame}

\begin{frame}{CDDFuse -- Correlation-Driven Loss}
    \small
    \textbf{(a) Correlation Loss cho Base Features} -- base feature hai modality phải tương quan cao:
    \vspace{-0.1cm}
    \begin{equation}
        \mathcal{L}_{\text{base}} = 1 - \rho(F_{\text{base}}^{IR},\, F_{\text{base}}^{VIS}), \quad \rho(X,Y) = \frac{\text{Cov}(X,Y)}{\sigma_X \cdot \sigma_Y}
    \end{equation}

    \vspace{0.1cm}
    \textbf{(b) Decorrelation Loss cho Detail Features} -- detail feature hai modality phải \textbf{ít tương quan}:
    \vspace{-0.1cm}
    \begin{equation}
        \mathcal{L}_{\text{detail}} = |\rho(F_{\text{detail}}^{IR},\, F_{\text{detail}}^{VIS})|
    \end{equation}

    \vspace{0.1cm}
    \begin{block}{Tổng Loss}
        $\mathcal{L}_{\text{total}} = \lambda_1 \mathcal{L}_{\text{base}} + \lambda_2 \mathcal{L}_{\text{detail}} + \lambda_3 \mathcal{L}_{\text{recon}} + \lambda_4 \mathcal{L}_{\text{perception}}$
    \end{block}
\end{frame}

\begin{frame}{CDDFuse -- Quá trình Fusion}
    \textbf{Ý tưởng:} Base features correlate cao $\to$ fuse bằng weighted avg. Detail features decorrelate $\to$ fuse bằng max/concat để giữ tính riêng biệt.

    \vspace{0.3cm}
    \textbf{Fusion rules:}
    \begin{itemize}
        \item \textbf{Base fusion:}
        \begin{equation}
            F_{\text{base}}^{F} = \frac{F_{\text{base}}^{IR} + F_{\text{base}}^{VIS}}{2}
        \end{equation}
        \item \textbf{Detail fusion:}
        \begin{equation}
            F_{\text{detail}}^{F} = \alpha \cdot F_{\text{detail}}^{IR} + (1-\alpha) \cdot F_{\text{detail}}^{VIS}
        \end{equation}
        trong đó $\alpha$ \textbf{học từ attention module}
        \item \textbf{Reconstruction:} IDWT/decoder từ $(F_{\text{base}}^{F},\, F_{\text{detail}}^{F}) \to$ fused image
    \end{itemize}
\end{frame}

\subsection{Modality-Specific / Shared Decomposition}

\begin{frame}{Modality-Specific / Shared Decomposition}
    \small
    \textbf{Mục tiêu:} Tách \textbf{shared content} (cấu trúc giải phẫu) và \textbf{modality-specific content} (thermal, texture riêng).

    \vspace{0.1cm}
    Cho encoder $E_s$ (shared), $E_m$ (specific), decoder $D$:
    \vspace{-0.1cm}
    \begin{align}
        z_s^A = E_s(I^A),\; z_s^B = E_s(I^B) &\;\Rightarrow \text{giống nhau (shared)} \nonumber \\
        z_m^A = E_m(I^A),\; z_m^B = E_m(I^B) &\;\Rightarrow \text{khác nhau (specific)} \nonumber \\
        I^A = D(z_s^A,\, z_m^A) &\;\text{(reconstruction)}
    \end{align}

    \vspace{-0.1cm}
    \textbf{Loss disentanglement:}
    \begin{itemize}\setlength{\itemsep}{2pt}
        \item \textbf{Cross-reconstruction:} $I^{A \to B} = D(z_s^A, z_m^B)$ phải giống $I^B$
        \item \textbf{MK-MMD:} $\text{MMD}^2(X,Y) = \|\mu_X - \mu_Y\|^2_{\mathcal{H}}$
    \end{itemize}
\end{frame}

\begin{frame}{DAF-Net (2024) \& PSFusion (2023)}
    \textbf{DAF-Net (2024):}
    \begin{itemize}
        \item Dùng \textbf{Multi-Kernel MMD} trong base encoder để align latent space IR và VIS
        \item INN-based detail encoder tách riêng texture từng modality
        \item Cross-attention giữa shared và specific features trước khi fusion
    \end{itemize}

    \vspace{0.3cm}
    \textbf{PSFusion (2023) -- Semantic Layer Decomposition:}
    \begin{itemize}
        \item Phân tách theo semantic levels dựa trên độ sâu trong mạng backbone (ResNet)
        \item Layer nông (early) $\to$ low-level feature (cạnh, màu sắc)
        \item Layer sâu (late) $\to$ high-level semantic feature (object, class)
    \end{itemize}
    \begin{align}
        F_l &= E_l(I), \quad l = 1..L \nonumber \\
        F^{\text{fused}} &= \sum_l w_l \cdot \text{Fusion}(F_l^{IR},\, F_l^{VIS})
    \end{align}
\end{frame}

\begin{frame}{Transformer-CNN Hybrid (DAF-Net, 2024)}
    \textbf{Mô hình hybrid} dùng Transformer và CNN \textbf{song song}:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Transformer branch:} self-attention trên patch $\to$ global feature $\to$ \textbf{base component}
        \item \textbf{CNN branch:} conv layers $\to$ local feature $\to$ \textbf{detail component}
        \item \textbf{Cross-branch fusion:} cross-attention giữa hai nhánh để bổ sung thông tin
    \end{itemize}

    \vspace{0.3cm}
    \begin{block}{Ý nghĩa}
        Kết hợp ưu điểm của Transformer (global context) và CNN (local detail) để tạo ra phân tách base-detail hiệu quả hơn.
    \end{block}
\end{frame}

\section{Nhóm 3: Generative / Diffusion-based Decomposition}
% ============================================================

\begin{frame}{Nhóm 3: Diffusion-based -- Tổng quan}
    \textbf{Đặc điểm:} Không phân tách ảnh theo cách truyền thống mà formulate fusion như bài toán \textbf{conditional generation} hoặc \textbf{Bayesian inference} trong không gian latent của mô hình sinh.

    \vspace{0.3cm}
    \textbf{Các phương pháp chính:}
    \begin{enumerate}
        \item DDPM -- Denoising Diffusion Probabilistic Models
        \item DDFM -- Fusion như Bayesian Posterior Sampling (ICCV 2023)
        \item Dif-Fusion -- Frequency-domain Diffusion (2023)
        \item VLM-Guided Fusion (ICML 2024)
    \end{enumerate}
\end{frame}

\subsection{DDPM}

\begin{frame}{DDPM -- Lý thuyết nền tảng}
    \small
    \textbf{Forward process:} Thêm Gaussian noise vào $x_0$ qua $T$ bước:
    \vspace{-0.15cm}
    \begin{equation}
        q(x_t | x_{t-1}) = \mathcal{N}\!\left(x_t;\; \sqrt{1-\beta_t}\, x_{t-1},\; \beta_t \mathbf{I}\right)
    \end{equation}
    \vspace{-0.25cm}
    \textbf{Closed form:} $q(x_t | x_0) = \mathcal{N}\!\left(x_t;\; \sqrt{\bar{\alpha}_t}\, x_0,\; (1-\bar{\alpha}_t)\, \mathbf{I}\right)$, \; với $\bar{\alpha}_t = \prod_{s=1}^{t} (1 - \beta_s)$

    \vspace{0.15cm}
    \textbf{Reverse process:} Học loại noise để tái tạo ảnh:
    \vspace{-0.15cm}
    \begin{equation}
        p_\theta(x_{t-1}|x_t) = \mathcal{N}\!\left(x_{t-1};\; \mu_\theta(x_t,t),\; \Sigma_\theta(x_t,t)\right)
    \end{equation}
    \vspace{-0.2cm}
    \textbf{Training Loss:}
    \vspace{-0.1cm}
    \begin{equation}
        \mathcal{L}_{\text{DDPM}} = \mathbb{E}_{x_0, \epsilon, t}\!\left[\left\|\epsilon - \epsilon_\theta\!\left(\sqrt{\bar{\alpha}_t} x_0 + \sqrt{1-\bar{\alpha}_t}\,\epsilon,\; t\right)\right\|^2\right]
    \end{equation}
\end{frame}

\subsection{DDFM (ICCV 2023)}

\begin{frame}{DDFM -- Fusion như Bayesian Posterior Sampling}
    \small
    \textbf{DDFM} (ICCV 2023 Oral) -- fusion như tính posterior:
    \vspace{-0.15cm}
    \begin{equation}
        p(x_F | I_{IR}, I_{VIS}) \;\propto\; p(x_F) \cdot p(I_{IR}, I_{VIS} | x_F)
    \end{equation}
    \vspace{-0.2cm}
    Phân ra bằng \textbf{EM algorithm}:
    \begin{itemize}\setlength{\itemsep}{2pt}
        \item \textbf{Subproblem 1:} $\nabla_{x_t} \log p_\theta(x_t)$ -- score function của pre-trained DDPM (\textbf{không cần re-train})
        \item \textbf{Subproblem 2:} $\nabla_{x_t} \log p(I_{IR}, I_{VIS} | x_t)$ -- hierarchical Bayesian:
        \vspace{-0.1cm}
        \begin{equation}
            \log p(I | x_t) = \sum_i \log p(I_i | z_i) + \log p(z_i | x_t)
        \end{equation}
        \vspace{-0.3cm}
        \footnotesize $z_i$: latent variable, tối ưu bằng EM (E-step estimate $z_i$, M-step update gradient).
    \end{itemize}

    \vspace{0.1cm}
    \begin{block}{\footnotesize Đặc điểm}
        \footnotesize \textbf{Ưu:} Zero-shot, không cần training data. \textbf{Nhược:} Inference chậm ($T$ denoising steps).
    \end{block}
\end{frame}

\begin{frame}{Dif-Fusion \& VLM-Guided Fusion}
    \textbf{Dif-Fusion (2023):}
    \begin{itemize}
        \item Phân tách ảnh theo \textbf{frequency domain} trước khi áp dụng diffusion
        \item Low-frequency fusion bằng \textbf{weighted average}
        \item High-frequency enhance bằng \textbf{guided diffusion process}
        \item Bảo toàn color fidelity và tăng sharpness đồng thời
    \end{itemize}

    \vspace{0.3cm}
    \textbf{VLM-Guided Fusion (ICML 2024):}
    \begin{itemize}
        \item Sử dụng Vision-Language Model (CLIP, BLIP-2) để hướng dẫn decomposition và fusion
        \item Text prompt (ví dụ: \textit{``preserve pedestrian thermal signature''}) được encode thành text embedding
        \item Cross-attend với image feature để điều khiển fusion weight
        \item Decomposition trở nên \textbf{task-adaptive}
    \end{itemize}
\end{frame}

\section{Nhóm 4: Physical-Prior Decomposition}
% ============================================================

\subsection{Retinex Decomposition}

\begin{frame}{Retinex Decomposition -- Lý thuyết}
    \textbf{Retinex} (Land \& McCann, 1971) mô hình hóa ảnh quan sát $I$ như \textbf{tích} của hai thành phần:
    \begin{equation}
        I(x,y) = L(x,y) \cdot R(x,y)
    \end{equation}
    \begin{itemize}
        \item $L(x,y)$: \textbf{illumination map} -- ánh sáng nền, biến đổi chậm
        \item $R(x,y)$: \textbf{reflectance map} -- nội dung thực, màu sắc và cạnh
    \end{itemize}

    \vspace{0.2cm}
    Trong log domain: $\log I = \log L + \log R$

    \vspace{0.2cm}
    Illumination $L$ ước lượng bằng \textbf{Total Variation minimization}:
    \begin{equation}
        L^* = \arg\min_L \left\{\|\log I - \log L\|^2 + \lambda \cdot TV(\log L)\right\}
    \end{equation}
\end{frame}

\begin{frame}{Retinex -- Ứng dụng trong Low-light Image Fusion}
    \textbf{Bài toán:} IR có thông tin thermal reliable, cao tương phản. VIS có thông tin màu sắc.

    \vspace{0.3cm}
    \textbf{Sau Retinex decomposition:}
    \begin{itemize}
        \item Lấy illumination $L$ từ IR (IR có độ tương phản cao và nhất quán)
        \item Lấy reflectance $R$ từ VIS (VIS có thông tin màu sắc)
        \item Tái tạo:
    \end{itemize}
    \begin{equation}
        I_F = L_{IR} \cdot R_{VIS} \quad \text{(enhanced, colorful, high-contrast)}
    \end{equation}
\end{frame}

\subsection{Hilbert Vibration Decomposition (HVD)}

\begin{frame}{Hilbert Vibration Decomposition (HVD, 2024)}
    \small
    HVD là phương pháp \textbf{adaptive decomposition} không cần basis function cố định.

    \vspace{0.1cm}
    Tín hiệu $f(t)$ biểu diễn là tổng các thành phần dao động:
    \vspace{-0.1cm}
    \begin{equation}
        f(t) = \sum_k A_k(t) \cdot \cos(\phi_k(t)) + r(t)
    \end{equation}
    \vspace{-0.2cm}
    \footnotesize ($A_k(t)$: instantaneous amplitude, $\phi_k(t)$: instantaneous phase)

    \small
    \vspace{0.1cm}
    Phép biến đổi Hilbert tính analytical signal:
    \vspace{-0.1cm}
    \begin{equation}
        f_a(t) = f(t) + i \cdot \mathcal{H}\{f(t)\} = A(t) \cdot e^{i\phi(t)}
    \end{equation}

    \vspace{-0.1cm}
    \begin{block}{\footnotesize So sánh với EMD}
        \footnotesize HVD \textbf{không dùng sifting process} nên ít mode mixing. Decompose ảnh 2D thành spatial frequency components \textbf{non-stationary}.
    \end{block}
    \vspace{0.05cm}
    \footnotesize \textbf{Ưu:} Không cần cut-off frequency, thích nghi local structure. Phù hợp ảnh texture phức tạp.
\end{frame}

\section{Nhóm 5: INN / Flow-based}
% ============================================================

\begin{frame}{Nhóm 5: INN / Flow-based -- Tổng quan}
    Sử dụng tính \textbf{khả nghịch (invertibility)} như ràng buộc kiến trúc, đảm bảo \textbf{không có thông tin nào bị mất}.

    \vspace{0.2cm}
    $\Rightarrow$ Quan trọng trong y tế (interpretability, reconstruction).
\end{frame}

\subsection{Nền tảng toán học của INN}

\begin{frame}{Affine Coupling Layer (RealNVP)}
    \textbf{RealNVP} (Dinh et al., 2017) -- Affine Coupling Layer:

    \vspace{0.2cm}
    Cho input $\mathbf{x} = (x_1, x_2)$:
    \begin{align}
        y_1 &= x_1 \nonumber \\
        y_2 &= x_2 \odot \exp(s(x_1)) + t(x_1)
    \end{align}
    Trong đó $s(\cdot)$ và $t(\cdot)$ là các mạng neural tùy ý (\textbf{không cần khả nghịch}).

    \vspace{0.2cm}
    \textbf{Biến đổi ngược:}
    \begin{align}
        x_1 &= y_1 \nonumber \\
        x_2 &= (y_2 - t(y_1)) \odot \exp(-s(y_1))
    \end{align}

    \vspace{0.2cm}
    Jacobian là triangular $\Rightarrow$ determinant tính dễ dàng: $\log|{\det J}| = \sum s(x_1)$
\end{frame}

\begin{frame}{Change of Variables \& Haar Wavelet Downsampling}
    \textbf{Change of Variables Formula:}

    Với INN $f: x \to z$, likelihood của dữ liệu:
    \begin{equation}
        \log p(x) = \log p_Z(f(x)) + \log \left|\det\!\left(\frac{\partial f}{\partial x}\right)\right|
    \end{equation}
    Trong đó $p_Z$ là prior (Gaussian). Training \textbf{maximize log-likelihood} này.

    \vspace{0.3cm}
    \textbf{Haar Wavelet Downsampling trong INN:}

    Thay vì strided convolution (mất thông tin), nhiều INN dùng Haar wavelet downsampling để squeeze spatial dimensions vào channel dimension:
    \begin{equation}
        x \in \mathbb{R}^{H \times W \times C} \to z \in \mathbb{R}^{H/2 \times W/2 \times 4C}
    \end{equation}
    Đây là phép biến đổi \textbf{khả nghịch}, preserve thông tin qua tất cả layers.
\end{frame}

\subsection{FusionINN (2024)}

\begin{frame}{FusionINN -- End-to-end Invertible Fusion}
    FusionINN tích hợp đồng thời image decomposition và fusion vào một \textbf{normalizing flow end-to-end}:
    \begin{equation}
        (z_s,\, z_A,\, z_B) = f_{INN}(I^A,\, I^B)
    \end{equation}
    Trong đó $z_s$ là shared latent, $z_A, z_B$ là modality-specific latents.

    \vspace{0.2cm}
    \textbf{Fused image:}
    \begin{equation}
        I^F = f_{INN}^{-1}(z_s,\, 0,\, 0)
    \end{equation}

    \textbf{Tái tạo source images:}
    \begin{equation}
        \hat{I}^A = f_{INN}^{-1}(z_s,\, z_A,\, 0)
    \end{equation}

    \vspace{0.2cm}
    \begin{exampleblock}{Ý nghĩa y tế}
        Khả năng tái tạo source images từ fused image là đặc tính quan trọng trong ứng dụng y tế.
    \end{exampleblock}
\end{frame}

\begin{frame}{FusionINN -- Training Objective \& EMMA}
    \textbf{Training Objective:}
    \begin{itemize}
        \item \textbf{Reconstruction loss:} $\|I^A - \hat{I}^A\|_1 + \|I^B - \hat{I}^B\|_1$
        \item \textbf{Fusion quality:} perceptual loss + gradient loss trên $I^F$
        \item \textbf{Disentanglement:} KL divergence align $z_s$; khuyến khích $z_A, z_B$ orthogonal
    \end{itemize}

    \vspace{0.3cm}
    \textbf{EMMA -- Equivariant Multi-Modal Alignment (CVPR 2024):}
    \begin{itemize}
        \item Sử dụng \textbf{equivariant network} để học phép biến đổi giữa modalities
        \item Equivariance đảm bảo: nếu ảnh bị transform (rotate, flip), feature map cũng transform tương ứng
        \item $\Rightarrow$ Alignment tốt hơn giữa IR và VIS khi có misalignment
    \end{itemize}
\end{frame}

\section{Kết luận}
% ============================================================

\begin{frame}{Kết luận}
    \small
    Image decomposition đóng vai trò \textbf{nền tảng} trong Image Fusion SOTA.

    \vspace{0.15cm}
    \textbf{Các nhận định chính:}
    \begin{itemize}\setlength{\itemsep}{3pt}
        \item \textbf{Không có phương pháp tối ưu toàn diện:} Multi-scale nhanh; DL chất lượng hơn; Diffusion tốt nhất visual quality nhưng chậm.
        \item \textbf{Frequency split (low/high) là mẫu số chung} của hầu hết framework.
        \item \textbf{INN và invertibility ngày càng quan trọng} -- đặc biệt trong y tế.
        \item \textbf{Diffusion model mở paradigm mới:} Fusion như Bayesian inference, zero-shot.
        \item \textbf{Xu hướng 2025:} Mamba + Wavelet + Task-aware loss.
    \end{itemize}
