\section{Không gian màu RGB}
% ============================================================

\begin{frame}{1. Không gian màu RGB (Red, Green, Blue)}
    \textbf{Đặc điểm cơ bản:}
    \begin{itemize}
        \item Là mô hình màu \textbf{cộng tính} (additive color model).
        \item Dựa trên sự kết hợp của ba màu cơ bản: Đỏ (Red), Lục (Green), Lam (Blue).
        \item Được sử dụng phổ biến trong các thiết bị hiển thị (màn hình, TV) và cảm biến ảnh (Camera).
        \item Mỗi thành phần thường được biểu diễn bằng 8 bit (giá trị từ 0 đến 255).
    \end{itemize}

    \vspace{0.3cm}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Biểu diễn hình học:}
            \begin{itemize}
                \item Một khối lập phương trong hệ tọa độ 3D.
                \item Gốc tọa độ $(0,0,0)$ là màu đen.
                \item Đỉnh $(255,255,255)$ là màu trắng.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \centering
            \includegraphics[width=0.8\textwidth]{Image/RGB.png}
            \captionof{figure}{Mô hình khối màu RGB}
        \end{column}
    \end{columns}
\end{frame}

\section{Không gian màu YUV}
% ============================================================

\begin{frame}{2. Không gian màu YUV}
    \textbf{Khái niệm:}
    \begin{itemize}
        \item Tách biệt thông tin độ sáng (\textbf{Luminance}) và thông tin màu sắc (\textbf{Chrominance}).
        \item \textbf{Y}: Độ sáng (Luma).
        \item \textbf{U, V}: Các thành phần hiệu màu (Chroma).
        \item Được phát triển để tương thích ngược giữa TV màu và TV trắng đen.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Ưu điểm:}
    \begin{itemize}
        \item Tận dụng đặc điểm mắt người: Nhạy cảm với độ sáng hơn màu sắc.
        \item Cho phép giảm băng thông truyền tải của các thành phần U, V mà không làm giảm đáng kể chất lượng cảm nhận (Chroma Subsampling).
    \end{itemize}
\end{frame}

\begin{frame}{Cấu trúc và Chuyển đổi YUV}
    Công thức chuyển đổi cơ bản từ RGB sang YUV (theo tiêu chuẩn BT.601):
    
    \begin{block}{Công thức chuyển đổi}
    \begin{equation}
    \begin{aligned}
        Y &= 0.299R + 0.587G + 0.114B \\
        U &= -0.147R - 0.289G + 0.436B = 0.492(B - Y) \\
        V &= 0.615R - 0.515G - 0.100B = 0.877(R - Y)
    \end{aligned}
    \end{equation}
    \end{block}

    \vspace{0.3cm}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item Thành phần $Y$ chứa thông tin chi tiết.
                \item Hiệu quả trong nén và truyền dẫn.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \centering
            \includegraphics[width=0.8\textwidth]{Image/yuv.jpg}
            \captionof{figure}{Tách kênh Y, U, V}
        \end{column}
    \end{columns}
\end{frame}

\section{Không gian màu YCrCb}
% ============================================================

\begin{frame}{3. Không gian màu YCrCb}
    \textbf{Đặc điểm:}
    \begin{itemize}
        \item Là một phiên bản rời rạc (digital) của YUV.
        \item Được sử dụng rộng rãi trong nén ảnh \textbf{JPEG} và nén video \textbf{MPEG, H.264}.
        \item \textbf{Y}: Luminance (Luma).
        \item \textbf{Cb}: Chroma Blue (Hiệu màu xanh lam: $B - Y$).
        \item \textbf{Cr}: Chroma Red (Hiệu màu đỏ: $R - Y$).
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Phạm vi giá trị (8-bit):}
    \begin{itemize}
        \item Y thường nằm trong khoảng [16, 235].
        \item Cb, Cr nằm trong khoảng [16, 240].
        \item Việc giới hạn này giúp tránh hiện tượng quá tín hiệu trong truyền dẫn.
    \end{itemize}
\end{frame}

\begin{frame}{Chuyển đổi RGB sang YCrCb}
    Công thức chuẩn cho tín hiệu kỹ thuật số (ITU-R BT.601):

    \begin{block}{Ma trận chuyển đổi}
    \begin{equation}
    \begin{bmatrix} 
        Y \\ Cb \\ Cr 
    \end{bmatrix} = 
    \begin{bmatrix} 
        16 \\ 128 \\ 128 
    \end{bmatrix} + 
    \begin{bmatrix} 
        65.481 & 128.553 & 24.966 \\
        -37.797 & -74.203 & 112.000 \\
        112.000 & -93.786 & -18.214
    \end{bmatrix}
    \begin{bmatrix} 
        R \\ G \\ B 
    \end{bmatrix}
    \end{equation}
    \end{block}
    \footnotesize{(Lưu ý: R, G, B ở đây thường được chuẩn hóa về khoảng [0, 1])}

    \vspace{0.3cm}
    \textbf{Lấy mẫu Chroma (Subsampling):}
    Cấu trúc 4:4:4, 4:2:2, 4:2:0 giúp tiết kiệm dung lượng.
    
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Chuyển đổi JPEG}
            \begin{equation}
            \begin{aligned}
                Cb &= (B - Y) / 2 + 0.5 \\
                Cr &= (R - Y) / 1.6 + 0.5
            \end{aligned}
            \end{equation}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \centering
            \includegraphics[width=0.8\textwidth]{Image/ycbcr.png}
            \captionof{figure}{Thang màu YCrCb}
        \end{column}
    \end{columns}
\end{frame}

\section{Không gian màu HSY}
% ============================================================

\begin{frame}{4. Không gian màu HSY (Hue, Saturation, Luma)}
    \textbf{Khái niệm:}
    \begin{itemize}
        \item Là một biến thể của mô hình HSV, sử dụng giá trị \textbf{Luma (Y)} thay cho cường độ đơn giản.
        \item \textbf{H (Hue)}: Tông màu (Góc màu $0^\circ - 360^\circ$).
        \item \textbf{S (Saturation)}: Độ bão hòa (Độ thuần khiết).
        \item \textbf{Y (Luma)}: Độ sáng cảm nhận (tổng trọng số RGB).
    \end{itemize}
\end{frame}

\begin{frame}{Tại sao dùng HSY/HSV?}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item Gần gũi với mô tả con người ("màu đỏ nhạt").
                \item Hữu ích trong tách ngưỡng (thresholding) theo màu.
                \item Ổn định hơn RGB dưới các điều kiện chiếu sáng khác nhau.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \centering
            \includegraphics[width=0.8\textwidth]{Image/HSV.png}
            \captionof{figure}{Tọa độ HSV}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Chi tiết về HSV (Hue - Saturation - Value)}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item \textbf{Hue}: Góc màu.
                \item \textbf{Saturation}: Độ bão hòa.
                \item \textbf{Value}: Độ sáng.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \centering
            \includegraphics[width=0.8\textwidth]{Image/HSV.png}
            \captionof{figure}{Mô hình nón HSV}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{So sánh HSY với RGB}
    \begin{itemize}
        \item \textbf{RGB}: Các thành phần có mối tương quan rất chặt chẽ. Thay đổi độ sáng thường làm thay đổi cả ba kênh.
        \item \textbf{HSY}: Các thành phần được tách biệt rõ rệt.
        \begin{itemize}
            \item Thay đổi \textbf{Y} giúp điều chỉnh độ sáng mà không làm biến đổi màu sắc (\textit{Color constant}).
            \item Thay đổi \textbf{S} giúp làm ảnh rực rỡ hơn hoặc chuyển về ảnh xám.
        \end{itemize}
    \end{itemize}

    \vspace{0.4cm}
    \begin{block}{Công thức Luma (Y)}
        \begin{equation}
            Y = 0.299R + 0.587G + 0.114B
        \end{equation}
        Giá trị này phản ánh đúng độ sáng mà mắt người cảm nhận được (màu xanh lá trông sáng hơn màu xanh dương ở cùng cường độ vật lý).
    \end{block}
\end{frame}
